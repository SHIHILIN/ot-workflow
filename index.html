function getFaqsFromEditor(id){const rows=document.querySelectorAll('#'+id+' .faq-row');const f=[];rows.forEach(function(r){const q=r.querySelector('.faq-row-q').value.trim();const a=r.querySelector('textarea').value.trim();if(q)f.push({q:q,a:a});});return f;}

function renderCards(){

  const container=document.getElementById('cardsContainer');

  const quickLinks=document.getElementById('quickLinks');

  container.innerHTML='';quickLinks.innerHTML='';

  const list=currentFilter==='all'?cardsData:cardsData.filter(function(c){return c.theme===currentFilter;});

  if(currentFilter!=='all'&&list.length>1){

    quickLinks.style.display='flex';

    list.forEach(function(card){

      const a=document.createElement('span');

      a.textContent=card.title;

      a.style.cssText='font-size:11px;color:#4a9eda;padding:3px 10px;background:rgba(74,158,218,0.1);border:1px solid rgba(74,158,218,0.3);border-radius:20px;cursor:pointer;white-space:nowrap;transition:all 0.2s;';

      a.addEventListener('mouseover',function(){this.style.background='rgba(74,158,218,0.2)';});

      a.addEventListener('mouseout',function(){this.style.background='rgba(74,158,218,0.1)';});

      a.addEventListener('click',function(){const t=document.getElementById('card-'+card.id);if(t)t.scrollIntoView({behavior:'smooth',block:'center'});});

      quickLinks.appendChild(a);

    });

  } else {quickLinks.style.display='none';}


  list.forEach(function(card){

    const faqHtml=(card.faqs&&card.faqs.length)?'<div class="faq-list">'+card.faqs.map(function(f){return'<div class="faq-item"><button class="faq-q">'+f.q+'<span class="faq-toggle">＋</span></button><div class="faq-a">'+f.a+'</div></div>';}).join('')+'</div>':'';

    let igDeepLink=null;

    if(card.theme==='technique'&&card.link){const m=card.link.match(/instagram\.com\/(?:reel|p|tv)\/([A-Za-z0-9_-]+)/);if(m)igDeepLink='instagram://media?id='+m[1];}

    const imgSrc=card.image||null;

    let imageBlock;

    if(imgSrc&&igDeepLink){imageBlock='<div class="card-image" style="cursor:pointer;position:relative;" data-ig="'+igDeepLink+'" data-url="'+card.link+'"><img src="'+imgSrc+'" alt="'+card.title+'"><div style="position:absolute;inset:0;display:flex;align-items:center;justify-content:center;background:rgba(0,0,0,0.3);border-radius:8px;"><div style="width:44px;height:44px;border-radius:50%;background:rgba(255,255,255,0.2);display:flex;align-items:center;justify-content:center;"><div style="width:0;height:0;border-top:9px solid transparent;border-bottom:9px solid transparent;border-left:15px solid white;margin-left:3px;"></div></div></div></div>';}

    else if(imgSrc){imageBlock='<div class="card-image"><img src="'+imgSrc+'" alt="'+card.title+'"></div>';}

    else{imageBlock='<div class="card-image">📌</div>';}

    let linkBlock='';

    if(card.link&&igDeepLink)linkBlock='<a class="card-link" data-ig="'+igDeepLink+'" data-url="'+card.link+'" href="#">▶ IG 影片 →</a>';

    const el=document.createElement('div');el.className='card';el.id='card-'+card.id;el.dataset.theme=card.theme;

    el.innerHTML=imageBlock+'<div class="card-title">'+card.title+'</div><div class="card-desc">'+card.description+'</div>'+faqHtml+linkBlock+'<div class="card-actions"><button class="btn edit" data-edit="'+card.id+'" style="flex:1">✏️ 編輯</button><button class="btn delete" data-del="'+card.id+'" style="flex:1">刪除</button></div>';

    container.appendChild(el);

  });


  container.addEventListener('click',function(e){

    const faqQ=e.target.closest('.faq-q');

    if(faqQ){const a=faqQ.nextElementSibling;a.classList.toggle('open');faqQ.querySelector('.faq-toggle').textContent=a.classList.contains('open')?'－':'＋';return;}

    if(igEl){e.preventDefault();openSocial(igEl.dataset.ig||igEl.dataset.url,igEl.dataset.url||igEl.dataset.fb);return;}

    const editBtn=e.target.closest('[data-edit]');

    if(editBtn){openEdit(parseInt(editBtn.dataset.edit));return;}

    const delBtn=e.target.closest('[data-del]');

    if(delBtn){if(confirm('確定要刪除嗎？')){cardsData=cardsData.filter(function(c){return c.id!==parseInt(delBtn.dataset.del);});renderCards();}return;}

  });

}

}
